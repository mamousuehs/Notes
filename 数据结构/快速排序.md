#include<iostream>
using namespace std;
const int N=1e6+10；
int n;
int q[N];
void quicksort(int q[];int l,int r){
    if(l>=r)return;
    int x=q[l+r>>1],i=l-1,j=r+1;
    while(i<j){
        do i++;while(q[i]<x);
        do j--;while(q[j]>x);
        if (i<j) swap(q[i],q[j]);
    }
    quicksort(q,l,j);
    quicksort(q,j+1,r);
}
int main(){
    scanf("%d",&n);
    for(int i=0;i<n;i++){
        scanf("%d",&q[i]);
    }
quicksort(q,0,n-1);

}

}

## 3.拆解
1. 找分界点`x`,一般就是`q[l]`,`q[r]`,`q[(l+r)/2]`。
2. 调整区间，根据分界点将小于等于`x`的放在左边，大于等于`x`的放在右边。
3. 递归
#### 如果啥都不记得可以用最朴素的方式
1. 开辟`a[]`,`b[]`两个空间
2. 逐个扫描，将小于等于`x`的放在`a[]`,大于`x`的放在`b[]`。
3. 把`a[]`,`b[]`放回`q[]`。
代码在[[快排笨比思想]]。
#### 双指针思想
1. 两个指针，从数列左右同时开始判断
2. 左边，若小于基准值就就右移一个位置，右边同理，大于就左移一个位置。
3. 不满足就停下，两边若都停下，就交换两个指针的所指向的数，这样就相当于又满足了。
4. 继续执行直到两个指针相遇结束，然后递归。
